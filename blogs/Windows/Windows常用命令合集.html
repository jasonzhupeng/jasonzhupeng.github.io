    Windows常用命令合集

    jason 1/9/2023 Windows

    🍉 windows常用到的各种命令合集

    # Win+R快捷方式

    services.msc  #服务管理
gpedit.msc  #策略编辑器
regedit     #注册表
control     #控制面板
inetmgr     #iis管理器，全称“inet manager”
control schedtasks      #任务计划程序
eventvwr.exe 或 `eventvwr`      #事件查看器，全称“event viewer”
compmgmt.msc  #计算机管理（磁盘管理、任务计划程序、事件查看器） 全称“computer management”
winver      #查看系统版本号
dxdiag      #查看系统型号
mstsc   #远程桌面连接
win+v   #打开剪贴板
win+x   #打开菜单列表

    # Windows服务管理

    sc create xx binpath="路径" #安装服务
sc queryex xx #查询服务状态
sc start xx #启动服务
sc stop xx #停止服务
sc delete xxx #删除服务

netstat -ano |findstr "LISTENING" #查看端口被占用情况
netstat -ano |findstr "0.0.0.0:80" #查看被占用端口关联的进程Id

tasklist |findstr 1136 #查看进程id所对应的exe程序
tasklist /fi "imagename eq nginx.exe" #根据程序名称查找程序
taskkill /f /im nginx.exe  #结束后台进程

    # Cmd Cli

    shutdown -s -t 10 #10秒后关机
shutdown -a #取消关机
shutdown -h #休眠
ping /? #测试ip是否能连接
telnel 119.23.233.64 80 #测试端口是否可用
tracert --ip /? #最终一个ip或域名经过的路由
