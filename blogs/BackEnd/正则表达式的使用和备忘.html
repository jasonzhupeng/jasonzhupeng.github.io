    # js和c#正则使用的区别

    前端js正则以//符号开头结束

    /\d/.test(1)
    1

    C#正则开头结束无需加//，字符串前加@""即可

    var str = "测试\t\n\r测试测试测试\t\n测试测试测试—";
var regex = new Regex(@"^\s+", RegexOptions.Singleline);
var res = regex.Replace(str, "xx");
Console.WriteLine(res);

    # 符号解析

    ^ 一组正则的开始
$ 一组正则的结束
? 正则出现0次或1次
+ 正则出行1次或多次
* 正则出行0次、1次或多次
[] 匹配某个字符
\s 匹配空格，换行，缩进等... 等同于\t\r\n


    /[.]+/ 包含.字符
/[.].*/ 匹配字符.后面的全部内容
/^x$/ 只能有x字符,且字符串长度只能为1
/^x+$/ or /^x*$/ 只能有x字符，字符串长度任意
/^[\sa-z]$/ 只能有空格和小写a-z字母，且字符串长度只能为1
/^[\sa-z]*$/ 只能有空格和小写a-z字母，字符串长度任意
/^[^a-z]*$/ 不能包含小写a-z字母，字符串长度任意
/^[a-z0-9]{4,16}$/ 只能有a-z小写字母和0-9数字，并且字符串长度不能小于4大于16

    # 常用表达式

    匹配空格、星号、filename(整体)

    /[\s\*]|(filename)/g

    $1匹配小括号里面的内容；\w+ 匹配数字、字母、下划线多个字符；.* 贪婪匹配，匹配任意字符任意次，.任意字符 *0次或无数次

    "./filename.vue".replace(/^\.\/(.*)\.\w+$/, '$1');  //输出结果："filename"
