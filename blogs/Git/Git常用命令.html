    Git常用命令

    jason 1/5/2023 Git

    🍉 工作中经常会用到git命令，其中经常用到的整理如下，方便忘记的时候查阅

    # 配置

     
     
     
     
     














    ssh-keygen -t rsa -C '配置的邮箱'  生成sshkey
-t 指定加密类型默认rsa，除此之外还有dsa、ecdsa、eddsa等
-C 注释，用于区分，可以填邮箱、用户名等
-b：秘钥长度，默认2048位，一般不用设置
ssh 需要输入密码，但是git bash窗口感觉无法输入其实是为了安全防止密码显示，直接按键盘回车就行。

git update-git-for-windows  更新git到最新版本
git config --list  查看所有配置
git config --global --list | grep alias 查找配置
git config --global --list  查看当前用户配置
git config --local --list 查看仓库配置
git config --system --list 查看系统配置
git config user.name 查看用户名
git config --global user.name "xxx" 设置用户名
git config --global log.date iso  设置日期显示格式为标准格式
git config --global gui.encoding utf-8  设置中文编码方式
# 设置别名
git config --global alias.lg=log --graph --pretty=format:'%Cred%h %C(yellow)%d%Creset %s  %Cgreen(%ci) %C(bold blue)%an'

    # 远程仓库设置

    git remote -v   查看远程地址
git remote set-url origin --url 切换远程仓库
git remote add origin git@github.com/jasonzhupeng/xxx.git  关联远程仓库（远程仓库已存在情况）

    # 提交代码

    git status  查看当前状态
git add .   提交代码到暂存
git commit -m "ss"  提交代码到本地仓库 --no-verify
git commit -a -m "ss"   提交暂存并提交到本地仓库
git commit --amend  修改提交注释，合并最新修改到最近提交
git push origin dev   提交代码到远程仓库
git push -u origin dev 向远程分支提交数据，适用于本地版本库关联了一个新的远程版本库
git Push -f origin dev 强制向远程推送数据，即使远程版本高于本地版本

    # 拉取代码

    git fetch origin dev 更新不合并代码
git pull origin dev 更新并合并代码
git pull origin dev --allow-unrelated-histories 更新并合并2个独立不相关的仓库
git pull origin dev --rebase  基于当前master版本上增加commit，不会产生分叉，相当于 git pull+git rebase

    # 合并代码

    git merge fetch_head  合并远程分支到本地
git merge --abort  取消合并
git rebase -i head~2  合并最近2个提交
git rebase -i commitid  commitid后一个版本到最新版本合并
git rebase dev  基于当前master版本上增加commit，不会产生分叉
git rebase --continue  --合并冲突，并继续
git rebase --abort 取消合并，退出

    # 撤销版本（代码）

    git checkout . 取消本地所有修改
git checkout xxx 取消本地某个文件修改
git restore xxx 取消本地某个文件修改
git restore --staged xxx  取消暂存区文件
git reset --hard origin/dev
git reset xxx  取消暂存区文件
git reset --hard commitid 回退到某个版本，不保留暂存和工作区的内容
git reset --soft commitid 回退到某个版本，保留暂存内容
git reset --mixed commited 回退到某个版本，保留本地修改内容
git reset 取消暂存区文件 相当于 git reset headg
git reset head^ 回退一个版本

    # 分支操作

    git branch 查看本地分支
git branch -r 查看远程分支
git branch -a 查看所有分支
git branch -vv 查看本地分支追踪的远程分支
git branch --set-upstream-to=origin/dev dev 设置本地分支关联远程分支
git branch xxx  新建分支
git branch -d xxx  删除分支
git checkout -b xxx 新建分支，并切换到新建的分支
git checkout xxx 切换分支
git checkout commitid -b xxx 从某个提交版本新建分支
git checkout master -b xxx 从某个分支下创建分支
git push origin --delete dev 删除远程分支
git branch -m new-branch-name  重命名分支名称

    # 查看日志

    gitk 可视化查看日志工具
git log --oneline 查看历史记录
git log xxx 查看某个文件的历史记录
git log origin/dev 查看远程仓库日志
git log branch1 ^branch2 查看branch1分支有，而branch2中没有的log
git log branch1..branch2 查看branch2中比branch1中多提交了哪些内容
git log --author=jason 根据作者过滤
git log --since="2022-11-10" 查询某个日期后的日志
git log --after="2022-08-01" --before="2022-11-01" 查询某个时间范围的日志
git log --grep '提交' 根据注释过滤
git log --graph 显示节点图
git log -g 附加显示reflog日志
git log --merge 查看合并冲突文件
git log --merges 查看合并日志
git log --no-merges 查看非合并日志
git log --stat -p 查看修改明细
git log --name-status 查看修改文件列表和状态
git log --all 查看所有日志
git log -p fetch_head 查看远程分支日志
git log dev..origin/dev   查看本地分支和远程分支的差异，适用于fetch的情况
git reflog 查看操作记录
git shortlog 按作者分类，显示每个作者的提交数量

    # 查看/对比差异

    git fsck --lost-found  找回已删除的stash
git show commitid 查找版本
git merge commitid 恢复版本
git blame xxx 追溯某个文件的修改历史，显示每行代码最后修改人
git diff master xxx 当前分支与master分支文件差异对比
git diff head xxx 比较当前版本和暂存区、工作区某个文件的差异
git diff xxx 比较暂存区和工作区某个文件的差异
git difftool head --commitid xxx 比较2个版本某个文件的差异

    # 合并分支

    git merge bugfix 合并分支
git merge --force bugfix 强制合并分支
git cherry-pick commitID bugfix 合并某个分支提交记录到当前分支
git cherry-pick --abort 取消合并
git checkout --patch dev *test.cs  只合并某个文件到当前分支

    # 标签操作

    git pull origin tag v0.3.0  获取远程指定标签
git fetch origin tag v0.*  获取远程最新标签
git show-ref --tag --查看标签
git ls-remote -t 查看远程标签
git tag -n --列出所有标签
git tag -a V1.2 -m 'release 1.2'  --创建标签，默认当前最新版本创建
git tag -a V1.2 -m 'release 1.2'  --从历史版本创建标签
git tag -d xxx 删除本地标签
git show v0.1.0 查看标签
git push origin --delete v1.0删除远程标签
git push origin --tags 推送本地所有标签到远程
git push origin v1.0 推送指定标签到远程

    # 暂存操作

    git stash -m "保存当前更改" 将当前修改暂存
git stash pop 恢复修改,并清除stash
git stash apply 恢复修改,不清除stash
git stash show -p stash@{0}显示stash改变的内容
git stash drop stash@{0} 删除stash
git stash clear 删除所有暂存
git stash list --date=iso 显示暂存的创建时间
git stash push [filepath] -m "xxx"  仅暂存某一个文件

    # 删除和取消追踪

    git rm xxx  删除文件
git clean -fd 删除未跟踪的文件，-f 只删除文件，-fd 连同目录一起删除
git rm -r --cached filename 取消文件的跟踪 -r 参数 递归取消文件夹下所有文件
git ls-files -v | grep ^h 查看忽略的跟踪文件列表
git update-index --assume-unchange filename 忽略跟踪文件
git update-index --no-assume-unchange filename 取消忽略跟踪文件
